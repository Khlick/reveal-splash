CSS Neon Glow and Animation


/* Neon glow effect */
.neon-glow {
  color: #00FFFF;
  text-shadow: 
    0 0 5px #00FFFF,
    0 0 10px #00FFFF,
    0 0 20px #00FFFF,
    0 0 30px #00FFFF,
    0 0 40px #00FFFF,
    0 0 50px #00FFFF,
    0 0 75px #00FFFF;
  animation: pulse 1.5s infinite alternate, flicker 2s infinite;
}

/* Pulse effect */
@keyframes pulse {
  0% {
    text-shadow: 
      0 0 5px #00FFFF,
      0 0 10px #00FFFF,
      0 0 20px #00FFFF,
      0 0 30px #00FFFF,
      0 0 40px #00FFFF,
      0 0 50px #00FFFF,
      0 0 75px #00FFFF;
    transform: scale(1);
  }
  100% {
    text-shadow: 
      0 0 10px #00FFFF,
      0 0 20px #00FFFF,
      0 0 30px #00FFFF,
      0 0 40px #00FFFF,
      0 0 50px #00FFFF,
      0 0 75px #00FFFF,
      0 0 100px #00FFFF;
    transform: scale(1.1);
  }
}

/* Flicker effect */
@keyframes flicker {
  0% { opacity: 1; }
  5% { opacity: 0.8; }
  10% { opacity: 1; }
  15% { opacity: 0.9; }
  20% { opacity: 1; }
  25% { opacity: 0.95; }
  30% { opacity: 1; }
  35% { opacity: 0.85; }
  40% { opacity: 1; }
  100% { opacity: 1; }
}
